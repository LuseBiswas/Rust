Structs in rust let you structure data together. Similar to objects in java

struct User{
    active: bool,
    username: String,
    email: String,
    sign_in_count: u64
}

fn main(){
    let user1 = User {
        active: true,
        username: String ::from("someuser"),
        email: String::from("email@example.com"),
        sign_in_count: 64,
    };
    println!("User 1 username:{:?}",user1.username)
}
